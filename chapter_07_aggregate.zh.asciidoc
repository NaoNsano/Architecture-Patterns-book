class Product:
    def __init__(self, sku: str, batches: List[Batch]):
        self.sku = sku  #<1>
        self.batches = batches  #<2>

    def allocate(self, line: OrderLine) -> str:  #<3>
        try:
            batch = next(b for b in sorted(self.batches) if b.can_allocate(line))
            batch.allocate(line)
            return batch.reference
        except StopIteration:
            raise OutOfStock(f"Out of stock for sku {line.sku}")

def add_batch(
    ref: str, sku: str, qty: int, eta: Optional[date],
    uow: unit_of_work.AbstractUnitOfWork,
):
    with uow:
        product = uow.products.get(sku=sku)
        if product is None:
            product = model.Product(sku, batches=[])
            uow.products.add(product)
        product.batches.append(model.Batch(ref, sku, qty, eta))
        uow.commit()


def allocate(
    orderid: str, sku: str, qty: int,
    uow: unit_of_work.AbstractUnitOfWork,
) -> str:
    line = OrderLine(orderid, sku, qty)
    with uow:
        product = uow.products.get(sku=line.sku)
        if product is None:
            raise InvalidSku(f"Invalid sku {line.sku}")
        batchref = product.allocate(line)
        uow.commit()
    return batchref

We've mentioned a few times that we were modeling with aggregates because we want

为什么我们在这里感到舒适。

首先,我们有目的地模拟我们的数据,以便我们能够实现一个单一的数据
向数据库查询,并单独更新,以维持我们的变化。 This
业绩往往比提出大量特别询问的系统要好得多。 In
我们往往发现,交易缓慢慢
随着软件的不断发展,变得更为复杂。

Second, our data structures are minimal and comprise a few strings and
integers per row. We can easily load tens or even hundreds of batches in a few
milliseconds.

第三,我们预计,目前每个产品只有20批。
一旦使用atch,我们就可以从我们的计算中扣除。 这意味着
that the amount of data we were fetching shouldntt get out of control over time.

If we  did  expect to have thousands of active batches for a product, we wed have
a 几种选择。 对我们来说,我们可以用洗load装在bat上
product. From the perspective of our code, nothing would change, but in the
背景,SQLAlchemy将通过我们的数据页面。 这将导致更多的
每种请求的行程都较小。 Because we need to find only a
只要具备我们秩序的充分能力,我们就可能工作得很好。

class Product:
    def __init__(self, sku: str, batches: List[Batch], version_number: int = 0):  #<1>
        self.sku = sku
        self.batches = batches
        self.version_number = version_number  #<1>

    def allocate(self, line: OrderLine) -> str:
        try:
            batch = next(b for b in sorted(self.batches) if b.can_allocate(line))
            batch.allocate(line)
            self.version_number += 1  #<1>
            return batch.reference
        except StopIteration:
            raise OutOfStock(f"Out of stock for sku {line.sku}")
